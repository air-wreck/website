import System.Random
import System.Random.Shuffle

determine_best_vendor :: RandomGen g => [String] -> g -> IO String
determine_best_vendor vendors =
  return . (!!0) . (shuffle' vendors $ length vendors)

main :: IO ()
main =
  newStdGen >>= (determine_best_vendor ["Eric", "Sam", "Rishi"]) >>= print
